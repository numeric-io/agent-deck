   ```go
   g, _ := errgroup.WithContext(context.Background())
   for _, inst := range instances {
       inst := inst  // capture
       g.Go(func() error {
           for j := 0; j < 5; j++ {
               if err := inst.UpdateStatus(); err != nil {
                   return err
               }
               time.Sleep(100 * time.Millisecond)
           }
           return nil
       })
   }
   require.NoError(t, g.Wait())
   ```

   ```go
   }
   ```
